

```shell
#chmod分配权限
#方式一
#chmod权限分配: chmod 用户 +/- 权限 file
chmod g+w file #文件file所有者所在组分配“写”权限
chmod o+x file #文件file所有者所在组之外的用户分配“执行”权限
chmod u-x file #文件file所有者去除执行权限
chmod u+x,g+w,g+x,o+w,o+x #三种用户同时增加权限(比较繁琐，可以参考方式二)
#这里的“用户”包括：u(文件所有者)、g(文件所有者同组用户)、o(文件所有者所在组之外的用户)、a(所有用户)
#这里的“权限”包括：r(读)、w(写)、x(执行)

#方式二
#chmod权限分配： chmod 数字权限 file
#rwx rwx rwx 用户数字表示： 421 421 421, 每种用户的权限都是各个数字相加，这里是 777，表示三种用户具有读写执行的权限
chmod 755 file #文件所有者“读写执行”、同组“读和执行”、其他用户“读和执行”

#chmod缺点：无法对多用户进行权限细化
```
