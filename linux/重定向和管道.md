

```shell
############重定向###############
#重定向标准输入"<"
cat < a.txt #将a.txt文件内容交由cat处理

#重定向标准输出“>”(覆盖写入)、“>>”追加写入,将结果输出到文件中
echo hello world >a.txt #将正确运行结果输出到文件a.txt中

#重定向标准错误“2>”、“2>>”,仅将错误结果输出到文件
xxx 2>a.txt #指令不存在的错误将输出到a.txt

#标注输出的完整写法
echo helo world 1>a.txt #这与1使用可被省略

#脚本执行同时输出正确和错误信息(覆盖输出)
test.sh &>a.txt #只允许覆盖输出正确和错误的信息(仅支持 &>)

#脚本执行追加输出正确和错误(不存在&>>的写法)
test.sh 1>>a.txt 2>&1 #输出正确的，并把错误结果通过通道1输出

```



```shell

#管道符“|”,将前一个命令的结果交给后续命令
#实例：查看剩余内存大小
free -m|grep -i mem|awk '{print $4}'
```
